def cp(string):                                               #Q6
    string=string.lower()
    lis=list(string)

            lis.remove(" ")

def merge(d1,d2):                                     #Q9
    for i in d2:
        try:
            if d1[i]:
                d1[i]+=d2[i]
        except:
            d1[i]=d2[i]
    return d1
d1 = {'a': 10, 'b': 20}
d2 = {'b': 5, 'c': 15}
print("Output: "+str(merge(d1, d2)))

def find_anagrams(word, words):
    word_list=list(word)
    word_list=sorted(word_list)
    result=[]
    for i in words:
        if word_list==sorted(list(i)):
            result.append(i)
    return result

print("Output: ", find_anagrams(word, words))

import matplotlib.pyplot as plt

def normalize(lis):
    max_num = max(lis)
    min_num = min(lis)
    return [(x - min_num) / (max_num - min_num) for x in lis]

numbers = [1,10,20,30,40,50,60]

normalized_list = normalize(numbers)
print("Normalized numbers: "+str(normalized_list))

plt.plot(normalized_list)
plt.xlabel('No.of')
plt.ylabel('Normalized Value')
plt.show()
